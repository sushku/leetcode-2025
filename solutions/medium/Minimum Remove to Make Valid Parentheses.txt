// Title: Minimum Remove to Make Valid Parentheses
            // Difficulty: Medium
            // Language: Python3
            // Link: https://leetcode.com/problems/minimum-remove-to-make-valid-parentheses/

class Solution:
    def minRemoveToMakeValid(self, s: str) -> str:
        openP, closeP = 0, 0
        newS = ""
        for c in s:
            if c == "(":
                openP += 1
            if c == ")":
                openP -= 1
            if openP < 0:
                openP = 0
            else:
                newS += c
        for c in newS[::-1]:
            if c == ")":
                closeP += 1
            if c == "(":
                closeP -= 1
            if closeP < 0:
                closeP = 0
            else:
        res = ""
                res += c
        return res[::-1]
