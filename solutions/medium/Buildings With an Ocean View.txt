// Title: Buildings With an Ocean View
            // Difficulty: Medium
            // Language: Python3
            // Link: https://leetcode.com/problems/buildings-with-an-ocean-view/

class Solution:
    def findBuildings(self, heights: List[int]) -> List[int]:
        maxH = 0
        res = []
        for i in range(len(heights)-1, -1, -1):
            if heights[i] > maxH:
                res.append(i)
                maxH = heights[i]
        return res[::-1]
