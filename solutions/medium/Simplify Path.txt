// Title: Simplify Path
            // Difficulty: Medium
            // Language: Python3
            // Link: https://leetcode.com/problems/simplify-path/

class Solution:
    def simplifyPath(self, path: str) -> str:
        folders = path.split("/")
        stack = []
        for folder in folders:
            if folder in ["", "."]:
                continue
            if folder == "..":
                if stack:
                    stack.pop()
                continue
            stack.append(folder)
        return "/" + "/".join(stack)
