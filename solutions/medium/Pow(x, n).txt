// Title: Pow(x, n)
            // Difficulty: Medium
            // Language: Python3
            // Link: https://leetcode.com/problems/pow(x,-n)/

class Solution:
    def myPow(self, x: float, n: int) -> float:
        if n == 0:
            return 1
        if n == 1:
            return x
        if n < 0:
            n = -1 * n
            x = 1 / x
        if n % 2:
            return self.myPow(x * x, n // 2) * x
        else:
            return self.myPow(x * x, n // 2)
