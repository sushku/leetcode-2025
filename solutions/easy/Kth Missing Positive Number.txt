// Title: Kth Missing Positive Number
            // Difficulty: Easy
            // Language: Python3
            // Link: https://leetcode.com/problems/kth-missing-positive-number/

class Solution:
    def findKthPositive(self, arr: List[int], k: int) -> int:
        rem = k
        prev = 0
        for num in arr:
            if num - prev > rem:
                return prev + rem
            rem -= num - prev - 1
            prev = num
        return num + rem
