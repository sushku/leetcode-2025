// Title: Valid Parentheses
            // Difficulty: Easy
            // Language: Python3
            // Link: https://leetcode.com/problems/valid-parentheses/

class Solution:
    def isValid(self, s: str) -> bool:
        stack = []
        pMap = {")": "(", "}": "{", "]": "["}
        for p in s:
            if p in "({[":
                stack.append(p)
            else:
                if not stack or pMap[p] != stack.pop():
                    return False
        return len(stack) == 0
