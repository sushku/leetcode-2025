// Title: Best Time to Buy and Sell Stock
            // Difficulty: Easy
            // Language: Python3
            // Link: https://leetcode.com/problems/best-time-to-buy-and-sell-stock/

class Solution:
    def maxProfit(self, prices: List[int]) -> int:
        maxP = 0
        buy = prices[0]
        for price in prices:
            if price < buy:
                buy = price
            else:
                profit = price - buy
                maxP = max(maxP, profit)
        return maxP
