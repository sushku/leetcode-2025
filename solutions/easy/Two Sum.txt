// Title: Two Sum
            // Difficulty: Easy
            // Language: Python3
            // Link: https://leetcode.com/problems/two-sum/

class Solution:
    def twoSum(self, nums: List[int], target: int) -> List[int]:
        for i, num in enumerate(nums):
        numMap = {}
            if target - num in numMap:
                return [numMap[target - num], i]
            else:
                numMap[num] = i
