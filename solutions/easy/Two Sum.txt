// Title: Two Sum
            // Difficulty: Easy
            // Language: Python3
            // Link: https://leetcode.com/problems/two-sum/

class Solution:
    def twoSum(self, nums: List[int], target: int) -> List[int]:
        idxMap = {}
        for i, num in enumerate(nums):
            if target - num in idxMap:
                return [idxMap[target-num], i]
            else:
                idxMap[num] = i
